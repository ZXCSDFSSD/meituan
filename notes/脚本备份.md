(async function hyperStablePicker() {
    const TARGET_YEAR = "2026";
    const TARGET_MONTH = "02"; // 目标月份
    const START_DAY = "01";
    const END_DAY = "02";

    console.log(`🚀 启动强力探测模式: ${TARGET_YEAR}-${TARGET_MONTH}-${START_DAY}...`);

    // 1. 全域穿透查找 (已验证稳定)
    function findAnchor(doc) {
        const containers = Array.from(doc.querySelectorAll('.saas-form-item-control-wrapper'));
        let target = containers.find(c => c.querySelector('input[placeholder="开始日期"]'));
        if (target) return { container: target, doc };
        const iframes = doc.querySelectorAll('iframe');
        for (let f of iframes) {
            try {
                const res = findAnchor(f.contentDocument || f.contentWindow.document);
                if (res) return res;
            } catch (e) {}
        }
        return null;
    }

    const anchorInfo = findAnchor(document);
    if (!anchorInfo) return console.error("❌ 找不到输入框容器");
    const { container, doc: targetDoc } = anchorInfo;
    const input = container.querySelector('input[placeholder="开始日期"]');

    // 2. 异步等待元素出现的探测器
    function waitForElement(dateStr, timeout = 3000) {
        return new Promise((resolve, reject) => {
            const start = Date.now();
            const timer = setInterval(() => {
                // 策略：同时寻找精确 title 和内部文本
                const cell = targetDoc.querySelector(`td[title*="${dateStr}"] .saas-picker-cell-inner`) || 
                             Array.from(targetDoc.querySelectorAll('.saas-picker-cell-in-view .saas-picker-cell-inner'))
                                  .find(el => el.innerText.trim() === dateStr.split('-').pop().replace(/^0/, ''));

                if (cell && cell.getBoundingClientRect().width > 0) {
                    clearInterval(timer);
                    resolve(cell);
                } else if (Date.now() - start > timeout) {
                    clearInterval(timer);
                    reject(new Error(`探测超时: 找不到 ${dateStr}`));
                }
            }, 100);
        });
    }

    // 3. 模拟全套点击
    const powerClick = (el) => {
        ['mousedown', 'mouseup', 'click'].forEach(n => 
            el.dispatchEvent(new MouseEvent(n, { bubbles: true, cancelable: true }))
        );
    };

    try {
        // --- 第一步：强力唤醒 ---
        input.focus();
        powerClick(input);
        await new Promise(r => setTimeout(r, 500));

        // --- 第二步：年份/月份切换 ---
        // 先确保在 2025 年 1 月 (复用你之前的点击年/月按钮逻辑)
        const yearBtn = targetDoc.querySelector('.saas-picker-year-btn');
        if (yearBtn && !yearBtn.innerText.includes(TARGET_YEAR)) {
            yearBtn.click();
            await new Promise(r => setTimeout(r, 400));
            const yCell = await waitForElement(TARGET_YEAR);
            powerClick(yCell);
            await new Promise(r => setTimeout(r, 400));
        }

        const monthBtn = targetDoc.querySelector('.saas-picker-month-btn');
        if (monthBtn && !monthBtn.innerText.includes('1月')) {
            monthBtn.click();
            await new Promise(r => setTimeout(r, 400));
            const mCell = await waitForElement("-01"); // 匹配月份
            powerClick(mCell);
            await new Promise(r => setTimeout(r, 500));
        }

        // --- 第三步：探测并点击 01 号 ---
        console.log("🔍 正在探测 01 号...");
        const startDayStr = `${TARGET_YEAR}-01-01`;
        const startCell = await waitForElement(startDayStr);
        powerClick(startCell);
        console.log("✅ 01 号点击成功");

        // --- 第四步：缓冲并探测 20 号 ---
        await new Promise(r => setTimeout(r, 800)); 
        console.log("🔍 正在探测 20 号...");
        const endDayStr = `${TARGET_YEAR}-01-20`;
        const endCell = await waitForElement(endDayStr);
        powerClick(endCell);
        console.log("✨ 任务全流程完成！");

    } catch (err) {
        console.error("运行中断: " + err.message);
        console.log("💡 建议：请手动点开日期面板，确保年份显示为 2025，然后再运行脚本。");
    }
})();

        if (row) {
            const checkbox = row.querySelector('.sun-checkbox-input');
            const wrapper = row.querySelector('.sun-checkbox-wrapper');

            if (wrapper && wrapper.classList.contains('sun-checkbox-wrapper-checked')) {
                console.log(`✨ 门店 ${id} 之前已勾选过`);
                return true;
            }

            if (checkbox) {
                checkbox.click();
                console.log(`✅ 成功勾选门店: ${id}`);
                return true;
            }
        }
        return false;
    };

        console.log(`🔍 正在定位: ${id}...`);

async function changeDateRange(page, startDate, endDate, log) {
    log(`📅 设置日期范围: ${startDate} 至 ${endDate}`);

    try {
        await page.waitForSelector('input[placeholder="请选择日期"]', { timeout: 10000 });

        await page.evaluate((startDate, endDate) => {
            const setDate = (input, value) => {
                input.focus();
                input.value = value;
                input.dispatchEvent(new Event('input', { bubbles: true }));
                input.dispatchEvent(new Event('change', { bubbles: true }));
                input.blur();
            };
            const dateInputs = document.querySelectorAll('input[placeholder="请选择日期"]');
            if (dateInputs.length >= 2) {
                setDate(dateInputs[0], startDate);
                setDate(dateInputs[1], endDate);
            }
        }, startDate, endDate);

        log('✅ 日期已设置');
        await sleep(500);
        return true;

    } catch (e) {
        log(`❌ 设置日期失败: ${e.message}`);
        return false;
    }
}
